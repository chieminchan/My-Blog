# 判断是否是电话号码

function isPhone(phone) {
    // ^x表示以x开头，$表示结尾，[xyz]表示xyz中的任意一个，{n}表示重复n次
    const phoneReg = /^1[35789]\d{9}$/;
    return phoneReg.test(phone);
}
