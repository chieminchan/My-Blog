var addStrings = function(num1, num2) {
    const len1 = num1.length;
    const len2 = num2.length;

    if (!len1 && !len2) {
        return 0;
    }

    let i = len1 - 1;
    let j = len2 - 1;

    let carry = 0;
    let result = "";
    while (i >= 0 || j >= 0) {
        let number1 = i >= 0 ? +num1[i] : 0;
        let number2 = j >= 0 ? +num2[j] : 0;
        let sum = number1 + number2 + carry;
        result = (sum % 10) + result;
        carry = Math.floor(sum / 10);

        i--;
        j--;
    }

    if (carry) {
        result = carry + result;
    }

    return result;
};
